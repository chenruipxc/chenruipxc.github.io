设定好信息，梳理思路并将代理传入

class BingCrawler():
def int(self): #定义一个新变量，int函数
self.BASE_URL = “https://bing.ioliu.cn/" #必应图片网址
self.TOTAL_PAGE = 124 #页数一共有124页
self.headers = {
“User - Agent”: “Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/81.0.4044.9 Safari/537.36”
} #用户头文件

获取响应

def spider(self):   #定义如何爬取某个网站
    session = requests.Session()  #请求session
    for index in range(1 ,self.TOTAL_PAGE+1):  #实现翻页,从第一页开始
        url = self.BASE_URL + f"/?p={index}"  #第几页的网址
        response = session.get(url=url, headers=self.headers)   #响应session，并且得到数据
        return response.content.decode("utf8")   #使数据编码为utf-8

解析整理响应内容

def parser(self, html):  #把程序文本转换成编译器内部的一种结构数据
    selector = etree.HTML(html)  #构造一个XPath解析对象并对HTML文本进行自动修正。
    url_list = selector.xpath("//div[@class='item']//div[@class='options']/a[2]/@href") #在网址中找到要提取的内容
    return url_list  #返回到url_list
